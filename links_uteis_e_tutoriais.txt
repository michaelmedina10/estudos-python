git clone https://github.com/pyenv/pyenv-update.git $(pyenv root)/plugins/pyenv-update

.bashrc
> export PYENV_ROOT="$HOME/.pyenv"
> command -v pyenv >/dev/null || export PATH="$PYENV_ROOT/bin:$PATH"
> eval "$(pyenv init -)"
